import Article from "@/components/Article";
import { GetStaticProps } from "next";
import { getSlugFromTitle, getReadTime } from "@/server-scripts/post";

export default function MDX({ readTime, children }) {
  return (
    <Article title={title} date={date} readTime={readTime}>
      {children}
    </Article>
  );
}

export const getStaticProps = (context) => {
  const slug = getSlugFromTitle(title);
  if (slug) {
    return { props: { readTime: getReadTime(slug) } };
  }
  return { props: { readTime: 0 } };
};
